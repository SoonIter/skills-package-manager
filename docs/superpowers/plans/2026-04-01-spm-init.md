> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `spm init` with interactive and non-interactive (`--yes`) flows to generate `skills.json` safely.

**Architecture:** Add a dedicated `initCommand` in `commands/` that owns file-exists checks and manifest creation. Keep `runCli` as thin parsing/dispatch, and put interactive UX + agent preset mapping into `cli/prompt.ts` so command logic stays testable. Use TDD with command tests first, then implementation, then CLI routing and docs.

**Tech Stack:** TypeScript, Node.js fs/promises, @clack/prompts, Rstest

```ts
import { access } from 'node:fs/promises'
import path from 'node:path'
import { writeSkillsManifest } from '../config/writeSkillsManifest'
import type { InitCommandOptions, SkillsManifest } from '../config/types'

async function assertManifestMissing(cwd: string): Promise<void> {
  const filePath = path.join(cwd, 'skills.json')
  try {
    await access(filePath)
    throw new Error('skills.json already exists')
  } catch (error) {
    if ((error as NodeJS.ErrnoException).code === 'ENOENT') {
      return
    }
    throw error
  }
}

function createDefaultManifest(): SkillsManifest {
  return {
    installDir: '.agents/skills',
    linkTargets: [],
    skills: {},
  }
}

export async function initCommand(options: InitCommandOptions): Promise<SkillsManifest> {
  await assertManifestMissing(options.cwd)

  const manifest = createDefaultManifest()
  await writeSkillsManifest(options.cwd, manifest)

  return manifest
}
```

```ts
import * as p from '@clack/prompts'

export const UNIVERSAL_AGENT_NAMES = [
  'Amp',
  'Antigravity',
  'Cline',
  'Codex',
  'Cursor',
  'Deep Agents',
  'Firebender',
  'Gemini CLI',
  'GitHub Copilot',
  'Kimi Code CLI',
  'OpenCode',
  'Warp',
] as const

export const ADDITIONAL_AGENT_TARGETS = [
  { label: 'Augment', path: '.augment/skills' },
  { label: 'IBM Bob', path: '.bob/skills' },
  { label: 'Claude Code', path: '.claude/skills' },
  { label: 'OpenClaw', path: 'skills' },
  { label: 'CodeBuddy', path: '.codebuddy/skills' },
  { label: 'Command Code', path: '.commandcode/skills' },
  { label: 'Continue', path: '.continue/skills' },
  { label: 'Cortex Code', path: '.cortex/skills' },
  { label: 'Trae', path: '.trae/skills' },
] as const

export type InitPromptResult = {
  installDir: string
  linkTargets: string[]
}

export async function promptInitManifestOptions(): Promise<InitPromptResult> {
  const installDirInput = await p.text({
    message: 'Where should skills be installed?',
    initialValue: '.agents/skills',
  })

  if (p.isCancel(installDirInput)) {
    p.cancel('Initialization cancelled')
    process.exit(0)
  }

  const configureAdditional = await p.confirm({
    message: 'Do you want to configure additional agent link targets?',
    initialValue: false,
  })

  if (p.isCancel(configureAdditional)) {
    p.cancel('Initialization cancelled')
    process.exit(0)
  }

  let linkTargets: string[] = []
  if (configureAdditional) {
    p.note(UNIVERSAL_AGENT_NAMES.join('\n'), 'Universal (.agents/skills) — always included')

    const selected = await p.groupMultiselect({
      message: 'Which agents do you want to install to?',
      options: {
        'Additional agents': ADDITIONAL_AGENT_TARGETS.map((agent) => ({
          value: agent.path,
          label: `${agent.label} (${agent.path})`,
        })),
      },
      required: false,
    })

    if (p.isCancel(selected)) {
      p.cancel('Initialization cancelled')
      process.exit(0)
    }

    linkTargets = selected as string[]
  }

  const installDir = String(installDirInput).trim() || '.agents/skills'
  return { installDir, linkTargets }
}
```

```ts
import { promptInitManifestOptions } from '../cli/prompt'

export type InitPrompter = () => Promise<{ installDir: string; linkTargets: string[] }>

export async function initCommand(
  options: InitCommandOptions,
  promptInit: InitPrompter = promptInitManifestOptions,
): Promise<SkillsManifest> {
  await assertManifestMissing(options.cwd)

  const manifest = options.yes
    ? createDefaultManifest()
    : {
        ...(await promptInit()),
        skills: {},
      }

  await writeSkillsManifest(options.cwd, manifest)
  return manifest
}
```

```ts
import { initCommand } from '../commands/init'

export async function runCli(argv: string[], context?: { cwd?: string }) {
  const [, , command, ...rest] = argv
  const cwd = context?.cwd ?? process.cwd()

  if (command === 'init') {
    const { positionals, flags } = parseArgs(rest)

    if (positionals.length > 0) {
      throw new Error('init does not accept positional arguments')
    }

    for (const key of Object.keys(flags)) {
      if (key !== 'yes') {
        throw new Error(`Unknown flag for init: --${key}`)
      }
    }

    if ('yes' in flags && flags.yes !== 'true') {
      throw new Error('init --yes does not accept a value')
    }

    return initCommand({ cwd, yes: 'yes' in flags })
  }

  // existing add/install/update branches unchanged
}
```
